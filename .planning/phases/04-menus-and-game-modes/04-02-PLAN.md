**A. Add _add_noise function to asm/board.s**

Add before the `.subsections_via_symbols` line in board.s:

```asm
// ----------------------------------------------------------------------------
// _add_noise(w0 = num_rows)
//
// Fill the bottom num_rows of the board with random garbage blocks.
// Each row has at least 1 gap (random empty column). Other cells have
// a 50% chance of being filled with a random piece color (1-7).
//
// Precondition: board is empty (called after _reset_board, before _spawn_piece)
// Clamps num_rows to 0-19 (must leave at least 1 empty row for spawning).
//
// Uses callee-saved: x19=num_rows, x20=board_base, x21=current_row,
//                    x22=gap_col, x23=col_counter
// Stack: 64 bytes
// ----------------------------------------------------------------------------
.globl _add_noise
.p2align 2
_add_noise:
    stp     x29, x30, [sp, #-64]!
    stp     x19, x20, [sp, #16]
    stp     x21, x22, [sp, #32]
    stp     x23, x24, [sp, #48]
    mov     x29, sp

    mov     w19, w0                 // w19 = num_rows to fill
    cbz     w19, Lnoise_done        // 0 rows = nothing to do
    cmp     w19, #19
    b.le    Lnoise_clamp_ok
    mov     w19, #19                // clamp to 19 max
Lnoise_clamp_ok:

    adrp    x20, _board@PAGE
    add     x20, x20, _board@PAGEOFF  // x20 = board base

    // Fill from row 19 (bottom) upward, filling w19 rows total
    mov     w21, #19                // w21 = current row (start at bottom)

Lnoise_row:
    // Check if we've filled enough rows
    mov     w8, #19
    sub     w8, w8, w21             // rows_filled = 19 - current_row
    cmp     w8, w19
    b.ge    Lnoise_done

    // Choose random gap column: arc4random_uniform(10)
    mov     w0, #10
    bl      _arc4random_uniform
    mov     w22, w0                 // w22 = gap column

    // Fill 10 columns of this row
    mov     w23, #0                 // col = 0
Lnoise_col:
    cmp     w23, w22
    b.eq    Lnoise_skip             // this is the gap, leave empty

    // 50% chance of placing a block
    mov     w0, #2
    bl      _arc4random_uniform
    cbz     w0, Lnoise_skip         // 0 = no block

    // Random piece color: arc4random_uniform(7) + 1
    mov     w0, #7
    bl      _arc4random_uniform
    add     w0, w0, #1              // cell value 1-7

    // Store in board[row * 10 + col]
    mov     w8, #10
    mul     w8, w21, w8
    add     w8, w8, w23
    uxtw    x8, w8
    strb    w0, [x20, x8]
    b       Lnoise_next_col

Lnoise_skip:
    // Leave cell as 0 (empty)
Lnoise_next_col:
    add     w23, w23, #1
    cmp     w23, #10
    b.lt    Lnoise_col

    sub     w21, w21, #1            // move up one row
    b       Lnoise_row

Lnoise_done:
    ldp     x23, x24, [sp, #48]
    ldp     x21, x22, [sp, #32]
    ldp     x19, x20, [sp, #16]
    ldp     x29, x30, [sp], #64
    ret
```

**B. Add invisible mode to _lock_piece in asm/board.s**

After the `bl _clear_lines` call in _lock_piece (and before the epilogue), add a check for _opt_invisible. When enabled, iterate the board and set all non-zero cells to 8 (invisible marker). Cell value 8 is non-zero so _is_piece_valid still detects collisions, but _draw_board will skip rendering it.

```asm
    // Check invisible mode: if enabled, hide all locked cells
    adrp    x8, _opt_invisible@PAGE
    ldrb    w9, [x8, _opt_invisible@PAGEOFF]
    cbz     w9, Llock_skip_invisible

    // Set all non-zero board cells to 8 (invisible marker)
    adrp    x8, _board@PAGE
    add     x8, x8, _board@PAGEOFF
    mov     w9, #0
Llock_invis_loop:
    ldrb    w10, [x8, x9]
    cbz     w10, Llock_invis_next   // skip empty cells
    mov     w10, #8                 // invisible marker
    strb    w10, [x8, x9]
Llock_invis_next:
    add     w9, w9, #1
    cmp     w9, #200
    b.lt    Llock_invis_loop

Llock_skip_invisible:
```

Save the return value from `bl _clear_lines` (w0) in a callee-saved register BEFORE the invisible mode loop so it can be restored for the return. Use x24 (already saved in the 64-byte frame):

- Before `bl _clear_lines`: (already there)
- After `bl _clear_lines`: add `mov w24, w0` to save lines_cleared
- After the invisible loop (at Llock_skip_invisible label or if skipped): add `mov w0, w24`
- This ensures the return value is preserved through the invisible mode code.

**F. Wire _add_noise into game init in asm/main.s**

In the game initialization section of main.s (inside Lstate_game, after `bl _reset_board` and applying starting level, but BEFORE `bl _spawn_piece`), add:

**Important implementation notes:**
- Cell value 8 for invisible mode is intentional: _is_piece_valid treats any non-zero cell as occupied (collision), so invisible cells still block piece placement. Only _draw_board changes behavior for value 8.
- The invisible mode loop in _lock_piece sets ALL non-zero cells to 8, not just the newly locked piece. This matches the simplified approach from the research (immediate full-board invisibility after each lock).
- Ghost toggle only affects rendering (_draw_ghost_piece skip), not collision or ghost Y computation.
- Hold toggle only affects the input dispatch ('c' key ignored), not the hold panel rendering -- but since _hold_piece_type remains 0xFF when never used, the hold panel will show empty naturally.
